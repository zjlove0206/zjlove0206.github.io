"================
"Set 基础设置
"    包括基础配置的开启等(语法、文件)
"================
set nocompatible    "vim工作不需要考虑vi的兼容 nocp 简写
set nu              "打开文件显示行号 也可不显示 nonu
set ruler           "在右下角显示光标的坐标
set hlsearch        "高亮显示搜索结果
set incsearch       "边输入边搜索(实时搜索)
set showcmd         "在ruler左边显示当前正在输入的命令
set expandtab       "将tab键改为空格，默认是8个
set tabstop=4       "将tab键改为4个空格
set shiftwidth=4    "自动缩进时，使用4个空格，默认是8个
set cindent         "c/c++自动缩进
set mouse=          "设置鼠标滚轮非自然滚动
set fdm=indent      "代码折叠
set ic              "忽略大小写查找
set nofoldenable    "禁用vim折叠功能
set showmatch       "自动匹配括号
set backspace=indent,eol,start "设置退格删除功能

map 9 $             "通过9跳转到行末尾, 0默认跳转到行首
map <silent>  <C-A>  gg v G  "Ctrl-A 选中所有内容

syntax on           "自动语法高亮 enable 也可以，on可以覆盖当前对语法高亮的修改
filetype on         "启动文件类型检查
filetype plugin on  "运行vim加载文件类型插件
filetype indent on  "启动文件自动缩进

"=================
"Plugin 插件引入
"       Vundle 插件管理 Plugin & Bundle 均可
"       引入管理文件目录 vim/bundle/插件/doc/插件.txt 为插件的具体用法
"================
set rtp+=~/.vim/bundle/Vundle.vim
call vundle#begin()

Plugin 'VundleVim/Vundle.vim'           "插件管理
Plugin 'chxuan/change-colorscheme'      "颜色主题 F9 F10 切换主题
Plugin 'jiangmiao/auto-pairs'           "括号补全
Plugin 'scrooloose/nerdcommenter'       "快速注释
Plugin 'scrooloose/nerdtree'            "树形目录树
Plugin 'Valloric/YouCompleteMe'         "自动补全 Google工程师主研 neocomplete 这一款体验一般 日本货 
Plugin 'godlygeek/tabular'              "代码检测&格式化&对齐  ale 支持vim8.0 还有一个韩国牌子
Plugin 'ctrlpvim/ctrlp.vim'             "快速查找文件
Plugin 'vim-airline/vim-airline'        "装逼利器 各种导航线设置 负担会让vim略显卡顿
Plugin 'vim-airline/vim-airline-themes' "airline 主题
Plugin 'will133/vim-dirdiff'            "文件差异比对
Plugin 'haya14busa/incsearch.vim'       "极速跳转 据说是个日本卡哇伊搞得 sneak&easymotion 都可以搞搞
Plugin 'mhinz/vim-startify'             "显示最近使用的文件列表
Plugin 'Xuyuanp/nerdtree-git-plugin'    "git 目录树更改
Plugin 'Yggdroot/indentLine'            "代码缩进
Plugin 'tpope/vim-surround'             "方便的在字符两边插入各种符号 xml
Plugin 'tpope/vim-commentary'           "注释代码行 gcc gc 注释 gcu gcgc 撤销注释
Plugin 'shawncplus/phpcomplete.vim'     "php插件
Plugin 'vim-scripts/pdv-standalone'     "PHP文档注释
Plugin 'vim-scripts/progressbar-widget'

"Plugin 'L9'                              "VIM脚本库
"Plugin 'tpope/vim-fugitive'            "处理git merge 冲突 dv
"Plugin 'tpope/vim-repeat'              "重复执行命令
"Plugin 'tpope/vim-endwise'             "自动结束某些结构
"Plugin 'vim-scripts/a.vim'             "头文件&实现文件快速切换
"Plugin 'vim-scripts/DoxygenToolkit.vim' "快速注释
"Plugin 'vim-scripts/SQLComplete.vim'    "SQL补全

"html & css & js{}
"Plugin 'Valloric/MatchTagAlways'       "自动匹配html标签，支持在标签间跳转
"Plugin 'docunext/closetag.vim'         "自动闭合HTML标签
"Plugin 'gorodinskiy/vim-coloresque'    "正则语法高亮  主要针对css\sass\less
"Plugin 'othree/yajs.vim'               "js语法高亮
"Plugin 'othree/javascript-libraries-syntax.vim'
"Plugin 'rstacruz/sparkup', {'rtp': 'vim/'}

"tag tab 结构
"Plugin 'vim-scripts/taglist.vim'       "taglist 标记列表 代码结构概览
Plugin 'majutsushi/tagbar'             "标签结构

call vundle#end()            
filetype plugin indent on    

"===============
"多窗口操作 mac Command
"===============
map + <C-W>+          "+：扩大窗口
map - <C-W>-          "- :   缩小窗口
map <C-l> <C-W>l      "C-l 移动到右侧窗口
map <C-h> <C-W>h      "C-h 移动到左侧窗口
map <C-j> <C-W>j      "C-j 移动到下方窗口
map <C-k> <C-W>k      "C-k 移动到上方窗口

"===============
"不要交换文件和备份文件，减少冲突
"===============
set noswapfile
set nobackup
set nowritebackup

"===============
"改变主题 & 打开关闭目录树 & 设置默认主题 Jellybeans
"===============
map <F10> :NextColorScheme<CR>
imap <F10> <ESC> :NextColorScheme<CR>
map <F9> :PreviousColorScheme<CR>
imap <F9> <ESC> :PreviousColorScheme<CR>

map <F2> :NERDTreeToggle<CR>
imap <F2> <ESC> :NERDTreeToggle<CR>
"autocmd vimenter * NERDTree  "是否自动打开目录树

set background=dark
let g:solarized_termcolors=256
colorscheme jellybeans 

"===============
"taglist 配置信息
"===============
nmap <F3> <ESC>:TagbarToggle<RETURN>
set tags=tags
set autochdir
let g:tagbar_left = 0
let g:tagbar_width = 24
let g:tagbar_autoclose = 1
let g:tagbar_autofocus = 1
let g:tagbar_sort = 1

"===============
"代码缩进 标志线 默认为¦ Ctrl+i 控制显示 
"===============
let g:indentLine_char = '¦' 
"let g:indentLine_color_term = 239 
map <C-i> :IndentLinesToggle<CR> 

"================
"PHP定制内容以及php-doc 自动补全 ctrl+x ctrl+o
"================
autocmd FileType php noremap <C-L> :!php -l %<CR> "语法检查
autocmd FileType php set omnifunc=phpcomplete#CompletePHP

nnoremap <C-K> :call PhpDoc()<CR>
vnoremap <C-K> :call PhpDocRange()<CR>

" cpp_class_scope_highlight

"================
"airline配置 切换不同的buffer缓冲区

let g:airline_theme="molokai"

    let g:airline_symbols = {}

"let g:airline_symbols.notexists = '∄'
"let g:airline_symbols.whitespace = 'Ξ'
nnoremap <C-N> :bn<CR>
nnoremap <C-P> :bp<CR>
nmap <F6> <ESC>:w<RETURN>:b#<RETURN>

" ctrlp
let g:ctrlp_map = '<c-f>'
let g:ctrlp_cmd = ':CtrlP'
let g:ctrlp_working_path_mode = '0'
set wildignore+=*/tmp/*,*.so,*.swp,*.zip     " MacOSX/Linux

" incsearch.vim
map /  <Plug>(incsearch-forward)
map ?  <Plug>(incsearch-backward)
map g/ <Plug>(incsearch-stay)

"简易的跨服务器复制粘贴
